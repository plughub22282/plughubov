# Streak-система — план реализации

> **For agentic workers:** REQUIRED SUB-SKILL: используйте superpowers:subagent-driven-development (рекомендуется) или superpowers:executing-plans для пошагового исполнения. Шаги отмечены чекбоксами (`- [ ]`).

**Goal:** Считать подряд идущие дни захода в приложение и на порогах 3/7/28 давать пользователю выбор одной из двух наград (слот на бит / +1 к скачиваниям), с индикатором-огоньком в хедере и модалкой выбора.

**Architecture:** Вся авторитетная логика — в SECURITY DEFINER RPC Supabase (счётчик, награды, enforcement лимитов). Main-процесс (`src/main/streak.ts`) — тонкие обёртки над `supabase.rpc(...)` по образцу `src/main/referral.ts`. Renderer получает всё через `window.api.streak` (IPC), держит состояние в отдельном хуке `useStreak` (не расширяет `AuthState`), рисует индикатор в сайдбаре и модалку. Реализуется спека `docs/superpowers/specs/2026-07-10-streak-design.md`.

**Tech Stack:** Electron (main + preload + contextBridge), React + TypeScript + Tailwind (renderer), Supabase Postgres (plpgsql, SECURITY DEFINER RPC, RLS-триггер `prevent_role_change`).

## Global Constraints

- **Источник схемы БД — единый файл** `supabase/schema.sql` (папки `supabase/migrations/` нет). Новый SQL **дописывается в конец** файла; все `alter table` идемпотентны (`add column if not exists`), функции — `create or replace`. Применение к живой БД — вручную через Supabase SQL Editor / `psql`.
- **Паттерн SECURITY DEFINER RPC:** `language plpgsql`, `security definer`, `set search_path = public`; для записи в защищённые триггером `prevent_role_change` колонки — `perform set_config('app.allow_priv_change', 'on', true);` **до** `update public.profiles`; в конце — `revoke execute on function ... from public;` + `grant execute on function ... to authenticated;`.
- **Статус входа в renderer — литерал `'signedIn'`** (тип `UiAuthStatus` в `src/renderer/src/types.ts`). Не `signed_in`.
- **Wire-типы зеркалятся** (как у referral): интерфейсы результата определяются в `src/main/streak.ts` и **повторно** в `src/preload/index.ts`. `window.api` типизируется автоматически через `export type ElectronAPI = typeof api` (`src/preload/index.ts:463`) — отдельный `Api`-интерфейс вести не нужно.
- **IPC-хендлеры:** каждый начинается с `const blocked = rejectUntrustedSender(event); if (blocked) return blocked;`; ошибки заворачиваются в `toSafeError(e, '<рус. сообщение>', '<лог-префикс>')` (см. `src/main/referral.ts`).
- **UTC:** «календарный день» стрика считается по UTC (`(now() at time zone 'utc')::date`).
- **Тестового раннера в проекте нет** (нет vitest/jest). «Проверка» каждой TS-задачи = `npm run typecheck` (это `tsc -p tsconfig.node.json --noEmit && tsc -p tsconfig.web.json --noEmit`) и, где указано, `npm run build`. Проверка SQL-задач = применить фрагмент к dev-БД и выполнить приведённый `select`. Финальный ручной чек-лист — Task 11.
- **Коммиты частые:** один коммит на задачу (в конце каждой задачи).
- Ветку под работу создать заранее (не коммитить в `main`/`master` напрямую): `git checkout -b feat/streak`.

- [ ] **Step 1: Дописать `touch_streak()` в `supabase/schema.sql`**

```sql
-- Раз за сессию отмечает заход. Идемпотентно в рамках дня (diff=0 → no-op).
-- Не выдаёт награду сама — только взводит reward_pending на порогах 3/7/28.
create or replace function public.touch_streak()
  returns table (streak_count int, reward_pending boolean, reward_stage int)
  language plpgsql
  security definer
  set search_path = public
as $$
declare
  v_uid     uuid    := auth.uid();
  v_today   date    := (now() at time zone 'utc')::date;
  v_last    date;
  v_count   int;
  v_stage   int;
  v_pending boolean;
  v_diff    int;
begin
  if v_uid is null then
    return;
  end if;

  select p.streak_last_date,
         coalesce(p.streak_count, 0),
         coalesce(p.streak_reward_stage, 0),
         coalesce(p.streak_reward_pending, false)
    into v_last, v_count, v_stage, v_pending
    from public.profiles p
    where p.id = v_uid
    for update;

  if v_last is null then
    v_count := 1;
    v_stage := 0;
  else
    v_diff := v_today - v_last;
    if v_diff = 0 then
      -- Тот же день: ничего не меняем, отдаём текущее состояние.
      streak_count  := v_count;
      reward_pending := v_pending;
      reward_stage  := v_stage;
      return next;
      return;
    elsif v_diff = 1 then
      v_count := v_count + 1;
    else
      -- Пропуск дня — безусловный сброс.
      v_count := 1;
      v_stage := 0;
    end if;
  end if;

  -- Порог достигнут впервые в этом цикле → взводим pending.
  if v_count in (3, 7, 28) and v_stage < v_count then
    v_stage   := v_count;
    v_pending := true;
  end if;

  perform set_config('app.allow_priv_change', 'on', true);
  update public.profiles
    set streak_count          = v_count,
        streak_last_date      = v_today,
        streak_reward_stage   = v_stage,
        streak_reward_pending = v_pending
    where id = v_uid;

  streak_count  := v_count;
  reward_pending := v_pending;
  reward_stage  := v_stage;
  return next;
end;
$$;

revoke execute on function public.touch_streak() from public;
grant  execute on function public.touch_streak() to authenticated;
```

- [ ] **Step 2: Дописать `claim_streak_reward(p_choice text)`**

```sql
-- Выбор награды. Начисляет бонус текущего месяца, гасит pending.
-- На пороге 28 после выбора — стартует новый цикл (streak_count = 1).
create or replace function public.claim_streak_reward(p_choice text)
  returns table (streak_count int)
  language plpgsql
  security definer
  set search_path = public
as $$
declare
  v_uid     uuid := auth.uid();
  v_pending boolean;
  v_stage   int;
  v_count   int;
begin
  if v_uid is null then
    raise exception 'unauthorized';
  end if;
  if p_choice not in ('beat', 'download') then
    raise exception 'bad_choice';
  end if;

  select coalesce(p.streak_reward_pending, false),
         coalesce(p.streak_reward_stage, 0),
         coalesce(p.streak_count, 0)
    into v_pending, v_stage, v_count
    from public.profiles p
    where p.id = v_uid
    for update;

  if not v_pending then
    raise exception 'no_reward';
  end if;

  perform set_config('app.allow_priv_change', 'on', true);
  if p_choice = 'beat' then
    update public.profiles
      set bonus_beat_slots       = coalesce(bonus_beat_slots, 0) + 1,
          bonus_beat_slots_month = date_trunc('month', now())::date,
          streak_reward_pending  = false
      where id = v_uid;
  else
    update public.profiles
      set bonus_download_slots       = coalesce(bonus_download_slots, 0) + 1,
          bonus_download_slots_month = date_trunc('month', now())::date,
          streak_reward_pending      = false
      where id = v_uid;
  end if;

  if v_stage = 28 then
    update public.profiles
      set streak_count = 1, streak_reward_stage = 0
      where id = v_uid;
    v_count := 1;
  end if;

  streak_count := v_count;
  return next;
end;
$$;

revoke execute on function public.claim_streak_reward(text) from public;
grant  execute on function public.claim_streak_reward(text) to authenticated;
```

- [ ] **Step 1: Создать `src/main/streak.ts`**

```ts
import { ipcMain } from 'electron'
import { supabase } from './supabase'
import { rejectUntrustedSender } from './ipc-security'
import { toSafeError } from './errors'

/**
 * Streak-система. Вся авторитетная логика (счётчик дней, взвод награды, начисление
 * бонусов) — в SECURITY DEFINER функциях Supabase (touch_streak / claim_streak_reward,
 * см. supabase/schema.sql). Здесь — тонкие обёртки над RPC по образцу referral.ts.
 */

// ─── Wire-типы (зеркалятся в preload/index.ts) ────────────────────────────────

export interface StreakTouchResult {
  ok: boolean
  error?: string
  /** Текущая длина стрика (1..28). */
  streakCount?: number
  /** Достигнут порог, награда ещё не выбрана. */
  rewardPending?: boolean
  /** Какой порог обработан в текущем цикле. */
  rewardStage?: 0 | 3 | 7 | 28
}

export interface StreakClaimResult {
  ok: boolean
  error?: string
  streakCount?: number
}

function rpcRow<T>(data: unknown): T | undefined {
  return (Array.isArray(data) ? data[0] : data) as T | undefined
}

// ─── Операции ─────────────────────────────────────────────────────────────────

export async function touchStreak(): Promise<StreakTouchResult> {
  const { data: sess } = await supabase.auth.getSession()
  if (!sess.session) return { ok: false, error: 'Войдите, чтобы копить стрик.' }
  try {
    const { data, error } = await supabase.rpc('touch_streak')
    if (error) throw error
    const row = rpcRow<{ streak_count: number; reward_pending: boolean; reward_stage: number }>(data)
    return {
      ok: true,
      streakCount: row?.streak_count ?? 0,
      rewardPending: !!row?.reward_pending,
      rewardStage: (row?.reward_stage ?? 0) as 0 | 3 | 7 | 28
    }
  } catch (e) {
    return { ok: false, error: toSafeError(e, 'Не удалось обновить стрик.', '[Supabase] streak:touch error') }
  }
}

export async function claimStreakReward(choice: 'beat' | 'download'): Promise<StreakClaimResult> {
  if (choice !== 'beat' && choice !== 'download') {
    return { ok: false, error: 'Некорректный выбор награды.' }
  }
  const { data: sess } = await supabase.auth.getSession()
  if (!sess.session) return { ok: false, error: 'Войдите, чтобы забрать награду.' }
  try {
    const { data, error } = await supabase.rpc('claim_streak_reward', { p_choice: choice })
    if (error) throw error
    const row = rpcRow<{ streak_count: number }>(data)
    return { ok: true, streakCount: row?.streak_count ?? 0 }
  } catch (e) {
    return { ok: false, error: toSafeError(e, 'Не удалось забрать награду.', '[Supabase] streak:claim error') }
  }
}

// ─── IPC ────────────────────────────────────────────────────────────────────────

let ipcInitialized = false

/** Регистрирует IPC-каналы streak. Безопасно вызывать многократно. */
export function registerStreakIpc(): void {
  if (ipcInitialized) return
  ipcInitialized = true

  ipcMain.handle('streak:touch', (event) => {
    const blocked = rejectUntrustedSender(event)
    if (blocked) return blocked
    return touchStreak()
  })
  ipcMain.handle('streak:claim', (event, choice: 'beat' | 'download') => {
    const blocked = rejectUntrustedSender(event)
    if (blocked) return blocked
    return claimStreakReward(choice)
  })
}
```

```ts
import { useCallback, useEffect, useRef, useState } from 'react'
import type { UiAuthStatus } from '../types'

interface UseStreak {
  streakCount: number
  rewardPending: boolean
  rewardStage: 0 | 3 | 7 | 28
  /** Забрать награду выбранного типа. По успеху гасит rewardPending локально. */
  claim: (choice: 'beat' | 'download') => Promise<{ ok: boolean; error?: string }>
}

/**
 * Локальное состояние стрика. Отдельно от useAuth/AuthState — чтобы не растить и без
 * того 5x-продублированный интерфейс AuthState. Дёргает streak:touch ровно один раз,
 * когда статус впервые становится 'signedIn' (переход, не каждый рендер).
 */
export function useStreak(status: UiAuthStatus): UseStreak {
  const [streakCount, setStreakCount] = useState(0)
  const [rewardPending, setRewardPending] = useState(false)
  const [rewardStage, setRewardStage] = useState<0 | 3 | 7 | 28>(0)
  const touchedRef = useRef(false)

  useEffect(() => {
    if (status !== 'signedIn') {
      // Разлогинились — сбрасываем, чтобы повторный вход снова дёрнул touch.
      touchedRef.current = false
      setStreakCount(0)
      setRewardPending(false)
      setRewardStage(0)
      return
    }
    if (touchedRef.current) return
    touchedRef.current = true
    let active = true
    window.api.streak.touch().then((res) => {
      if (!active || !res.ok) return
      setStreakCount(res.streakCount ?? 0)
      setRewardPending(!!res.rewardPending)
      setRewardStage((res.rewardStage ?? 0) as 0 | 3 | 7 | 28)
    })
    return () => {
      active = false
    }
  }, [status])

  const claim = useCallback(async (choice: 'beat' | 'download') => {
    const res = await window.api.streak.claim(choice)
    if (res.ok) {
      setRewardPending(false)
      setStreakCount(res.streakCount ?? streakCount)
    }
    return { ok: res.ok, error: res.error }
  }, [streakCount])

  return { streakCount, rewardPending, rewardStage, claim }
}
```

**Примечание (уточнение спеки):** спека предлагала звать `consume_named_quota` с `p_limit` напрямую. Но `p_limit` тогда пришёл бы значением из кода-клиента, а бонус живёт на сервере. Поэтому вводим тонкую серверную RPC, которая сама считает `50 + bonus` (авторитетно) и переиспользует `consume_named_quota` внутри. - [ ] **Step 1: Дописать `consume_asset_download_quota()` в `supabase/schema.sql`**

```sql
-- Месячный лимит скачивания ассетов (preset/loop/drumkit). База free = 50,
-- premium — без лимита. Бонус стрика (bonus_download_slots этого месяца) поднимает
-- лимит. Лимит считается на сервере — клиент его не передаёт (в отличие от
-- consume_named_quota напрямую). Переиспользует usage_counters через consume_named_quota.
create or replace function public.consume_asset_download_quota()
  returns table (allowed boolean, used_after int, resets_at timestamptz)
  language plpgsql
  security definer
  set search_path = public
as $$
declare
  v_uid   uuid := auth.uid();
  v_bonus int  := 0;
  v_limit int;
begin
  if v_uid is null then
    return query select false, 0, now();
    return;
  end if;
  -- Premium — без лимита.
  if public.has_premium() then
    return query select true, 0, now();
    return;
  end if;
  select case when p.bonus_download_slots_month = date_trunc('month', now())::date
              then coalesce(p.bonus_download_slots, 0) else 0 end
    into v_bonus
    from public.profiles p
    where p.id = v_uid;
  v_limit := 50 + coalesce(v_bonus, 0);
  return query
    select * from public.consume_named_quota('asset_download_monthly', v_limit, interval '30 days');
end;
$$;

revoke execute on function public.consume_asset_download_quota() from public;
grant  execute on function public.consume_asset_download_quota() to authenticated;
```
